### Examples
```

  type "post"
  includes [
    friends: [
      :comments
    ],
    comments: []
  ]

  routes do
    base "/posts"

    get :read
    get :me, route: "/me"
    index :read

    patch :update
    related :comments, :read

    delete_from_relationship :comments

end
```
